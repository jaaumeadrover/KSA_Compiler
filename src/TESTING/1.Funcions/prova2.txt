func int sumarNombresArray(int array a, int b, int longitud){
    switch(b+1){
        case 2:
            b <- a[1];
        case false:
            b <- 3;
    }

    for(int i <- 0; i < longitud; i <- i + 1){
        b <- a[i] + b;
    }
    return b;
}

}
