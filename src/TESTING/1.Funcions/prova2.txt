func int sumarNombresArray(int array a, int b, int longitud){
    for(int i <- 0; i < longitud; i <- i + 1){
        b <- a[i] + b;
    }
    return b;
}

main(){
    //DECLARACIONS
    int l <- 10;
    int acum <- 0;

    int a1 <- int [10];
    for(int i <- 0; i < l; i <- i + 1){
        a1[i] <- 1;
    }
    //Cridem la funcio que ens sumarÃ  els nombres de l'array
    acum <- sumarNombresArray(a1,acum,l);
    print(acum);
}
